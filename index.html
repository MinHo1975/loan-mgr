🟡 STEP 3 — 아래 코드를 Apps Script에 붙여넣기
// ✅ 대출관리대장 - Google Apps Script 서버 코드
// Google Sheets 스프레드시트의 Apps Script에 이 코드 전체를 붙여넣으세요.

const SHEET_LOANS  = '대출기록';
const SHEET_TOOLDB = '공구DB';

function doGet(e) { return handleRequest(e); }
function doPost(e) { return handleRequest(e); }

function handleRequest(e) {
  var params = e.parameter || {};
  var postData = {};
  try { if (e.postData) postData = JSON.parse(e.postData.contents || '{}'); } catch(err) {}
  var action = params.action || postData.action || '';
  var result;
  try {
    if (action === 'ping')          result = { ok: true, msg: '연결 성공! Apps Script가 정상 작동 중입니다.' };
    else if (action === 'getAll')   result = getAllData();
    else if (action === 'saveAll')  result = saveAllData(postData);
    else result = { ok: false, msg: '알 수 없는 명령: ' + action };
  } catch(err) {
    result = { ok: false, msg: '서버 오류: ' + err.message };
  }
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function getOrCreateSheet(name, headers) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(name);
  if (!sh) {
    sh = ss.insertSheet(name);
    sh.getRange(1, 1, 1, headers.length).setValues([headers]);
    sh.setFrozenRows(1);
  }
  return sh;
}

var LOAN_HEADERS = ['id','dbRef','itemName','category','subCategory','spec','qty','unit',
  'borrower','contact','phone','loanDate','returnDueDate','permanent','purpose','remark',
  'partNumbers','returned','returnDate','printedAt','createdAt'];

var TOOL_HEADERS = ['id','name','category','subCategory','spec','maker','purchaseDate',
  'price','priceUnit','location','totalQty','qtyUnit','remark','partNumbers',
  'clients','createdAt'];

function getAllData() {
  var lsh = getOrCreateSheet(SHEET_LOANS,  LOAN_HEADERS);
  var dsh = getOrCreateSheet(SHEET_TOOLDB, TOOL_HEADERS);
  var loans  = sheetToObjects(lsh, LOAN_HEADERS);
  var toolDb = sheetToObjects(dsh, TOOL_HEADERS);
  return { ok: true, loans: loans, toolDb: toolDb };
}

function saveAllData(data) {
  var loans  = data.loans  || [];
  var toolDb = data.toolDb || [];
  var lsh = getOrCreateSheet(SHEET_LOANS,  LOAN_HEADERS);
  var dsh = getOrCreateSheet(SHEET_TOOLDB, TOOL_HEADERS);
  objectsToSheet(lsh, loans,  LOAN_HEADERS);
  objectsToSheet(dsh, toolDb, TOOL_HEADERS);
  return { ok: true, msg: '저장 완료 (대출:'+loans.length+'건, 공구DB:'+toolDb.length+'건)' };
}

function sheetToObjects(sh, headers) {
  var rows = sh.getDataRange().getValues();
  if (rows.length <= 1) return [];
  return rows.slice(1).map(function(row) {
    var obj = {};
    headers.forEach(function(h, i) {
      var v = row[i];
      if (h === 'partNumbers' || h === 'clients') {
        try { obj[h] = JSON.parse(v || '[]'); } catch(e) { obj[h] = []; }
      } else if (h === 'permanent' || h === 'returned') {
        obj[h] = (v === true || v === 'true');
      } else {
        obj[h] = (v === null || v === undefined) ? '' : String(v);
      }
    });
    return obj;
  }).filter(function(o){ return o.id; });
}

function objectsToSheet(sh, objects, headers) {
  var lastRow = sh.getLastRow();
  if (lastRow > 1) sh.getRange(2, 1, lastRow - 1, headers.length).clearContent();
  if (!objects.length) return;
  var rows = objects.map(function(obj) {
    return headers.map(function(h) {
      var v = obj[h];
      if (h === 'partNumbers' || h === 'clients') return JSON.stringify(v || []);
      if (v === null || v === undefined) return '';
      return v;
    });
  });
  sh.getRange(2, 1, rows.length, headers.length).setValues(rows);
}
🔴 STEP 4 — 저장 후 배포하기
1. 코드 붙여넣기 후 Ctrl+S (저장)
2. 우측 상단 배포 → 새 배포 클릭
3. ⚙️ 아이콘 → 웹 앱 선택
4. 설명: 아무 내용이나 입력 (예: v1)
5. 다음 사용자로 실행: 나 (본인 계정)
6. 액세스 권한: 모든 사용자
7. 배포 클릭 → 권한 요청 시 권한 검토 → 허용
8. 생성된 웹앱 URL을 복사
✅ STEP 5 — URL 입력 및 연결
1. 복사한 URL을 위 입력창에 붙여넣기
2. 연결 테스트 클릭 → 성공 메시지 확인
3. 저장 및 활성화 클릭
4. 동기화 버튼으로 데이터 업로드!
